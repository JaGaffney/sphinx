    def go_button(self, current_item, 
                        tab_name="tab4", 
                        explicit=False, 
                        recursive=False, 
                        action_mode=False, 
                        in_auto_run_action=None):
        mode = 0 # change
        #mode = 1 # pull 
        #mode = 2 # push
        # set the current item for convienance
        self.file_context.ftp_connect_list[tab_name].current_item = current_item

        #
        if (recursive==True):
            self.file_context.ftp_connect_list[tab_name].recursive = True
        else:
            self.file_context.ftp_connect_list[tab_name].recursive = False

        # check if directory
        if (current_item["file_type"] == "dir"):
            mode = 0
            if (self.file_context.ftp_connect_list[tab_name].recursive == True):
                 mode = 2
        elif (current_item["file"] == "cd.."):
            mode = 0
        else:
            mode = 1

        # mode
        self.log.write("mode:%s" % mode)

        directory = current_item["file"]
        
        #
        if (mode == 0 and action_mode != "remove"):
            self.file_context.ftp_connect_list[tab_name].changeDir(directory, explicit=explicit)

        if (action_mode == "remove"):
            self.file_context.ftp_connect_list[tab_name].trash_files_server(directory, in_callback=None, in_callback_file=None)
            self.clearLayout(tab_name)
            self.tabBrowserUI(tab_name=tab_name)
        elif (action_mode == "open"):
            file_wrapper = wrapper.Wrapper( in_context=self.context, 
                                            in_ftp=self.file_context.ftp_connect_list[tab_name], 
                                            in_schedule_manager=self.file_context.schedule_list[tab_name])
            file_wrapper.open()
        elif (action_mode == "publish"):
            passed_schedule = schedule_manager_ui.schedule_manager.copy_scheduleManager(self.context, self.file_context.schedule_list[tab_name])
            tmp_ftp = ftp_connect.copy_ftp(self.context,self.file_context.ftp_connect_list[tab_name])
            
            self.file_context.ftp_transfer_list.append(tmp_ftp)
            self.threading_form = threading_ftp.ThreadingMain(self.context, 
                                                              self.file_context.ftp_transfer_list[len(self.file_context.ftp_transfer_list)-1],
                                                              in_schedule_manager=passed_schedule)
            self.threading_form = set_scheme(self.threading_form, alt_style=False)
            self.threading_form.show()
        else:
            # only push pull files if they are of a valid status
            if (mode == 1):
                if (current_item["status"] != "green"):
                    tmp_ftp = ftp_connect.copy_ftp(self.context,self.file_context.ftp_connect_list[tab_name])
                    self.file_context.ftp_transfer_list.append(tmp_ftp)
                    self.threading_form = threading_ftp.ThreadingMain(self.context, self.file_context.ftp_transfer_list[len(self.file_context.ftp_transfer_list)-1])
                    self.threading_form = set_scheme(self.threading_form, alt_style=False)
                    self.threading_form.show()
            elif (mode == 2):
                tmp_ftp = ftp_connect.copy_ftp(self.context,self.file_context.ftp_connect_list[tab_name])
                self.file_context.ftp_transfer_list.append(tmp_ftp)
                self.threading_form = threading_ftp.ThreadingMain(self.context, self.file_context.ftp_transfer_list[len(self.file_context.ftp_transfer_list)-1], in_auto_run_action=in_auto_run_action)
                self.threading_form = set_scheme(self.threading_form, alt_style=False)
                self.threading_form.show()
            else:
                self.clearLayout(tab_name)
                self.tabBrowserUI(tab_name=tab_name)

    def on_context_menu(self, myListWidget, current_item, point, file=False, tab_name="tab4"):

        self.popMenu.exec_(myListWidget.mapToGlobal(point))

    def add_file_element(self, current_files, current):
        # only draw if not hidden
        my_item = None
        if (current_files[current]["hidden"] == False):
            my_item = QListWidgetItem(current)
            label_tips = ""
            for keys in list(current_files[current].keys()):
                label_tips += "%s = %s \n" % (keys, current_files[current][keys])
            my_item.setToolTip(label_tips)
            
            # change colour based on status
            cur_status = current_files[current]["status"]
            my_item.setBackground(QColor(25,25,25))
            if (cur_status == "red"):
                my_item.setForeground(QColor(250, 0, 0))
            elif (cur_status == "blue"):
                my_item.setForeground(QColor(42,130,218))
            elif (cur_status == "orange"):
                my_item.setForeground(QColor(218, 130, 42))
            elif (cur_status == "green"):
                my_item.setForeground(QColor(0, 250, 0))
        #
        return my_item

    def schedule_combo(self, tab_name="", active_list="", active_type="", active_value=""):
        combo = QComboBox()
        active_list.append("")
        for name in active_list:
            if name not in non_create_keys_ui:
                combo.addItem(name)
        #
        index = combo.findText(active_value, QtCore.Qt.MatchFixedString)
        if index >= 0:
            combo.setCurrentIndex(index)
        else:
            combo.setCurrentIndex(0)
        combo.currentIndexChanged.connect(partial(self.schedule_active_change, combo, tab_name=tab_name, active_type=active_type))
        return combo
