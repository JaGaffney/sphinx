    def __init__(self, url, parent=None, in_context=None, in_login=None):
        super(MainWindow, self).__init__(parent)


        # initlize our context

        self.menuBar = QtGui.QMenuBar(self)
        self.toolBar = QtGui.QToolBar(self)
        self.statusBar = QtGui.QStatusBar(self)


        # init ui
        self.tab_master_form_layout = QVBoxLayout()
        #self.statusBar = QtGui.QMenuBar(self)
        self.sb=self.statusBar

        self.pbar = QtGui.QProgressBar()
        self.pbar.setMaximumWidth(120)
        self.wb=QtWebEngineWidgets.QWebEngineView(loadProgress = self.pbar.setValue, loadFinished = self.pbar.hide, loadStarted = self.pbar.show, titleChanged = self.setWindowTitle)

        self.tb=self.toolBar

        for a in (QWebEnginePage.Back, QWebEnginePage.Forward, QWebEnginePage.Reload):
            self.tb.addAction(self.wb.pageAction(a))

        self.url = QtGui.QLineEdit(returnPressed = lambda:self.wb.setUrl(QtCore.QUrl.fromUserInput(self.url.text())))
        self.tb.addWidget(self.url)

        self.wb.urlChanged.connect(lambda u: self.url.setText(u.toString()))
        self.wb.urlChanged.connect(lambda: self.url.setCompleter(QCompleter(list([(i.url().toString()) for i in self.wb.history().items()]), caseSensitivity = QtCore.Qt.CaseInsensitive)))

        #self.wb.statusBarMessage.connect(self.sb.showMessage)
        self.wb.page().linkHovered.connect(lambda l: self.sb.showMessage(l, 3000))

        self.search = QtGui.QLineEdit(returnPressed = lambda: self.wb.findText(self.search.text()))
        self.search.hide()
        self.showSearch = QShortcut("Ctrl+F", self, activated = lambda: (self.search.show() , self.search.setFocus()))
        self.hideSearch = QShortcut("Esc", self, activated = lambda: (self.search.hide(), self.wb.setFocus()))

        self.quit = QShortcut("Ctrl+Q", self, activated = self.close)
        self.zoomIn = QShortcut("Ctrl++", self, activated = lambda: self.wb.setZoomFactor(self.wb.zoomFactor()+.2))
        self.zoomOut = QShortcut("Ctrl+-", self, activated = lambda: self.wb.setZoomFactor(self.wb.zoomFactor()-.2))
        self.zoomOne = QShortcut("Ctrl+=", self, activated = lambda: self.wb.setZoomFactor(self.wb.zoomFactor()+.2))

        self.sb.addPermanentWidget(self.search)
        self.sb.addPermanentWidget(self.pbar)
        self.wb.load(url)
        self.wb.loadFinished.connect(self._on_load_finished)
        # zoom in
        self.wb.setZoomFactor(self.wb.zoomFactor()+0.1)
        self.tab_master_form_layout.addWidget(self.tb)
        self.tab_master_form_layout.addWidget(self.sb)
        self.tab_master_form_layout.addWidget(self.wb)
        self.setLayout(self.tab_master_form_layout)
        self.setWindowTitle("Cinestron - Info")

def main():
    w = 2280/2; h = 1520/2
    app = QApplication(sys.argv)
    # set app icon    
    app_icon = QtGui.QIcon()
    logo_locaion = get_base_path() +'\images\logo.jpg'
    app_icon.addFile(logo_locaion, QtCore.QSize(16,16))
    app_icon.addFile(logo_locaion, QtCore.QSize(24,24))
    app_icon.addFile(logo_locaion, QtCore.QSize(32,32))
    app_icon.addFile(logo_locaion, QtCore.QSize(48,48))
    app_icon.addFile(logo_locaion, QtCore.QSize(256,256))
    app.setWindowIcon(app_icon)

    if len(sys.argv) > 1:
        url = QtCore.QUrl.fromUserInput(sys.argv[1])
    else:
        url = QtCore.QUrl('http://www.cinestron.com')
    ex=MainWindow(url)

    #ex = CivilizationUI()
    ex = set_scheme(ex)
    ex.show()
    ex.resize(w, h)

    sys.exit(app.exec_())

if __name__ == "__main__":
