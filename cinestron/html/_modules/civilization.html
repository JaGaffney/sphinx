def showdialog(in_context, in_terms_info):
    msg = QMessageBox()
    msg.setIcon(QMessageBox.Information)
    terms_info = in_terms_info
    msg.setText("Do you agree to the terms and conditions of using this software and servers?")
    msg.setInformativeText(terms_info)
    msg.setWindowTitle("Terms and conditions")
    #msg.setDetailedText(cinestron_user_info.terms_and_conditions)
    msg.setStandardButtons(QMessageBox.Yes | QMessageBox.No)

    retval = msg.exec_()
    return retval

        """

[docs]    def login_functions(self, name, password, passwordServer, passwordIM):
        """
        :type name: string
        :param name:
        :type password: string
        :param password:
        :type passwordServer:  string
        :param passwordServer:
        :type passwordIM:  string
        :param passwordIM:
        :rtype: None
        """
        # form login structure
        name = str(name)
        password = str(password)
        server_password = str(passwordServer)
        IM_password = str(passwordIM)

        # unified password
        if (self.login_mode == self.login_modes[0]):
            password = password
            server_password = password
            IM_password = password
            self.passwordServer.setText(server_password)    
            self.passwordIM.setText(IM_password)    

        # configure email
        email = name.replace("@cinestron.com", "")
        email += "@cinestron.com"

        self.context.user_info["user_name"] = name
        self.context.user_info["user_name_server"] = "X\\" + name
        self.context.user_info["user_email"] = email
        self.context.user_info["user_email_password"] = password
        self.context.user_info["user_server_password"] = server_password
        self.context.user_info["user_IM_password"] = IM_password

        sucessful_login = False
        # VALIDATE THE LOGIN
        validate_mode = "YANDEX"
        if (validate_mode == "YANDEX"):
            sucessful_login = valid_email(  fromaddr = email,
                                            frompassword = password,
                                            log=self.log)

        if (sucessful_login == True):
            try:
                self.context.user_info["fullname"] = cinestron_user_info.cinestron_user_info[name]["fullname"]
            except:
                self.context.user_info["fullname"] = self.context.user_info["user_name"]
            terms_info = cinestron_user_info.terms_and_conditions_simple
            terms_info = terms_info.replace("NAME", self.context.user_info["fullname"])

            choice = showdialog(self.context, terms_info)

            if choice == QMessageBox.Yes:
                sucessful_login = True
                self.email_signed_login(terms_info)
            else:
                sucessful_login = False

        if (sucessful_login == True):
            self.login_call()
        else:
            self.context.post_login = False
            self.username.setStyleSheet("color: rgb(200, 0, 0);")
            self.password.setStyleSheet("color: rgb(200, 0, 0);")
            self.passwordServer.setStyleSheet("color: rgb(200, 0, 0);")      
            self.passwordIM.setStyleSheet("color: rgb(200, 0, 0);")

        to_add = str(self.members_full_list.currentText())

        total_to = total_to.replace(",,", ",")
