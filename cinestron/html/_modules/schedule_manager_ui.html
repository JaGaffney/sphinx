[docs]    def delete_item(self, mode="", delete_mode="hidden"):
        """
        :type mode: string
        :param mode: mode for show/sequence/shot depending on which list has been clicked
        :rtype: None
        """
        # DATEBASE IN #
        # pull from server before modifing our result
        self.schedule_manager.show_json = self.schedule_manager.database_connection.db_get()

        if (self.schedule_manager.active_show != ""):
            mode="show"
        if (self.schedule_manager.active_sequence != ""):
            mode="sequence"
        if (self.schedule_manager.active_shot != ""):
            mode="shot"

        if (mode =="show"):
            # sets hidden to true for all nested keys seq/asset/shot
            if (delete_mode == "hidden"):
                self.schedule_manager.show_json[self.schedule_manager.active_show]["hidden"] = True
            elif (delete_mode == "pop"):
                try:
                    self.schedule_manager.show_json.pop(self.schedule_manager.active_show)
                except:
                    print("pop failed")
            #
            # DATEBASE OUT #
            # Push back to server after modifing our result
            self.schedule_manager.database_connection.db_push(self.schedule_manager.show_json)

            # these stop the program from crashing if a active show is not selected as well as making the ui cleaner.
            self.add_sequence.setEnabled(False)
            #self.add_sequence.setStyleSheet("color:" + CS_RED_STRING)
            self.add_shot.setEnabled(False)

            #
            self.refresh_show()

        elif (mode =="sequence"):
            # werid error where somtimes will crash, you need to go back to previous key in order to delete
            if self.schedule_manager.active_sequence !="":
                if (delete_mode == "hidden"):
                    self.schedule_manager.show_json[self.schedule_manager.active_show][self.schedule_manager.active_sequence]["hidden"] = True
                elif (delete_mode == "pop"):
                    try:
                        self.schedule_manager.show_json[self.schedule_manager.active_show].pop(self.schedule_manager.active_sequence)
                    except:
                        print("pop failed")
            #
            # DATEBASE OUT #
            # Push back to server after modifing our result
            self.schedule_manager.database_connection.db_push(self.schedule_manager.show_json)

            self.refresh_sequence(refresh_tree=False)
            self.schedule_manager.shot_listview_ui.clear()

        elif (mode =="shot"):
            if self.schedule_manager.active_shot !="":
                if (delete_mode == "hidden"):
                    self.schedule_manager.show_json[self.schedule_manager.active_show][self.schedule_manager.active_sequence][self.schedule_manager.active_shot]["hidden"] = True
                elif (delete_mode == "pop"):
                    try:
                        self.schedule_manager.show_json[self.schedule_manager.active_show][self.schedule_manager.active_sequence].pop(self.schedule_manager.active_shot)
                    except:
                        print("pop failed")
            #
            # DATEBASE OUT #
            # Push back to server after modifing our result
            self.schedule_manager.database_connection.db_push(self.schedule_manager.show_json)
            self.refresh_shot(refresh_tree=False)

[docs]    def delete_pop_up(self, mode="user"):
        """
        :type self:
        :param self: popup display for when a show is been selected for deleation
        :rtype: None
        """
        buttonReply = QMessageBox.question(self, 'Delete item', "Are you sure you want to permanently delete this item?", QMessageBox.Yes | QMessageBox.No, QMessageBox.No)
        if (buttonReply == QMessageBox.Yes):
            if mode == "admin":
                self.delete_item(delete_mode="pop")
            else:
                self.delete_item()
        else:
            if (self.schedule_manager.active_show != ""):
                self.schedule_manager.sequence_listview_ui.clear()
                self.schedule_manager.asset_listview_ui.clear()
            if (self.schedule_manager.active_sequence == ""):
                self.schedule_manager.sequence_listview_ui.clear()
                self.schedule_manager.asset_listview_ui.clear()
            if (self.schedule_manager.active_shot == ""):
                self.schedule_manager.shot_listview_ui.clear()



[docs]    def keyPressEvent(self, event):
        """
        :type event: event
        :param event: when a key is pressed
        :rtype: None
        """
        # is a pyqt function even though it doesnt say QT 
        if (event.key() == Qt.Key_Delete):
            if ( self.schedule_manager.active_show != "" or self.schedule_manager.active_sequence != "" or self.schedule_manager.active_shot != "" ):
                #
                self.delete_pop_up()
                self.schedule_manager.active_show = ""
                self.schedule_manager.active_sequence =""
                self.schedule_manager.active_shot = ""

        # when you press enter it will do the same features as add text button
        if (event.key() == Qt.Key_Return):
            if (self.my_show != ""):
                self.add_show_functions(self.my_show.text())
            if (self.sequence != ""):
                self.add_sequence_functions(self.sequence.text(), "default")
            if (self.asset != ""):
                self.add_sequence_functions(self.asset.text(), "asset")
            if (self.shot != ""):
                self.add_shot_functions(self.shot.text(), 
                                        self.schedule_manager.active_sequence)

[docs]    def on_context_menu(self, myListWidget, point, mode="show"):
        """
        :type myListWidget: object QWidget
        :param myListWidget: the current/active list that has been selected with a right click
        :type point: point?
        :param point: the current location of the mouse when a right click occurs
        :type mode: string
        :param mode: each list will have different options
        :rtype: None
        """
        item = myListWidget.currentItem()
        if item == None:
            return
        else:
            current = str(item.text())
      
        # add context buttons
        filebrowser_action = QAction('Open Filebrowser', self)
        sync_action = QAction('Sync to server', self)
        build_action = QAction('Build Folders (server)', self)
        build_local_action = QAction('Build Folders (local only)', self)
        delete_action = QAction('Delete', self)
        show_info_action = QAction('Show Info', self)
        rename_action = QAction('Rename', self)
        duplicate_action = QAction('Duplicate', self)

        # trigger events
        filebrowser_action.triggered.connect(lambda: self.open_file_browser(current))
        sync_action.triggered.connect(lambda:self.schedule_manager.ftpdb_push())
        build_action.triggered.connect(lambda: self.schedule_manager.create_folders_main_wrapper(build_folders=True, build_server_folders=True))
        build_local_action.triggered.connect(lambda:self.schedule_manager.create_folders_main_wrapper(build_folders=True, build_server_folders=False))
        delete_action.triggered.connect(partial(self.delete_pop_up))
        show_info_action.triggered.connect(lambda: self.add_show_info_functions(current))
        rename_action.triggered.connect(lambda: self.add_rename_functions(current))
        duplicate_action.triggered.connect(lambda: self.add_duplicate_function(current))

        # create context menu
        self.popMenu = QMenu(self)
        self.popMenu.addAction(filebrowser_action)
        self.popMenu.addSeparator()
        self.popMenu.addAction(sync_action)
        self.popMenu.addSeparator()
        self.popMenu.addAction(build_action)
        self.popMenu.addAction(build_local_action)

        self.popMenu.addSeparator()
        self.popMenu.addAction(rename_action)
        self.popMenu.addAction(delete_action)
        if (mode=="show"):
            self.popMenu.addSeparator()
            self.popMenu.addAction(show_info_action)
        else:
            self.popMenu.addAction(duplicate_action)
