[docs]    def build_sequence(self, in_show, in_sequence, in_mode="default"):
        """
        :type in_dict: dict
        :param in_dict: main dictionary
        :type in_show: string
        :param in_show: the current show key where the sequence will be added
        :type in_sequence: string 
        :param in_sequence: the current sequence key where the data will be created
        :rtype: dict
        """

        # DATEBASE IN #
        # pull from server before modifing our result
        in_dict = self.database_connection.db_get()

        s_show = in_dict[in_show]["show_name"]

        s_sequence = str(in_sequence)
        s_sequence = s_sequence.lower()
        s_sequence = s_sequence.replace(" ", "_")
        total_sequences = s_sequence.split(",")
        
        sequence_key = { "delivery_data": cinestron_task_info.delivery_data,
                         "external_data": cinestron_task_info.external_data,
                         "hidden": False,
                         "mode": in_mode}

        out_dict = copy.deepcopy(in_dict)
        
        for current in total_sequences:
            out_dict[s_show][current] = sequence_key

        # DATEBASE OUT #
        # Push back to server after modifing our result
        self.show_json = out_dict
        self.database_connection.db_push(self.show_json)

        return self.show_json

[docs]    def populate_list(self, in_dict, list_name):
        """
        :type in_dict: dict
        :param in_dict: the dictionary for what the list will be set with
        :type list_name: var
        :param list_name: the list name which will be populated
        :rtype: None
        """
        list_name.clear()
        list_keys = in_dict
        if (type(list_keys) is dict):    
            list_keys_sorted = sorted(list_keys.keys())
            for item_names_list in list_keys_sorted:
                my_item = QListWidgetItem(item_names_list)

                # skip the non ui keys
                if item_names_list in non_create_keys:        
                    continue
                if item_names_list in non_create_keys_ui:        
                    continue

                # don't display if hidden
                hidden = False
                val = in_dict[item_names_list]
                if (type(val) is dict): 
                    #in_dict[item_names_list]["hidden"] = False
                    if ("hidden" in list(val.keys())):
                        tmp_data = val["hidden"] 
                        if (tmp_data == True or tmp_data == "True" or tmp_data == "true"):
                            hidden = True

                if (hidden == False):
                    #
                    # labels
                    # we can manipulate the lables here further for better visibility on data
                    label_tips = "\n"
                    if (type(in_dict[item_names_list]) is dict): 
                        for keys in list(in_dict[item_names_list].keys()):
                            #
                            cur_data = in_dict[item_names_list][keys]
                            if (type(cur_data) is dict): 
                                if ("hidden" in list(cur_data.keys())):
                                    cur_data = "%s:%s" % ("hidden", cur_data["hidden"])

                            label_tips += "%s = %s \n" % (keys, cur_data)
                    else:
                        label_tips = "TODO define tooltip"
                    #
                    my_item.setToolTip(label_tips)
                    list_name.addItem(my_item)

[docs]    def find_in_obj(self, obj, condition, path=None):
        """
        :type obj: dict
        :param obj: full dict to search through
        :type condition: string
        :param condition: the filter string that you want to search through the dict
        :type path: string
        :param path: how deep into the dict you currently are based on previous keys
        :rtype: dict
        """
        if path is None:
            path = []    

        # In case this is a list
        if isinstance(obj, list):
            for index, value in enumerate(obj):
                new_path = list(path)
                new_path.append(index)
                for result in find_in_obj(value, condition, path=new_path):
                    yield result 

        # In case this is a dictionary
        if isinstance(obj, dict):
            for key, value in obj.items():
                new_path = list(path)
                new_path.append(key)
                for result in find_in_obj(value, condition, path=new_path):
                    yield result 

                if condition == key:
                    new_path = list(path)
                    new_path.append(key)
                    yield new_path, value
