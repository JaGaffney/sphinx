class PyAnalogClock(QWidget):
    """
    :type PyAnalogClock: class
    :param PyAnalogClock: displays time in analog format
    """
    timeChanged = QtCore.pyqtSignal(QtCore.QTime)

    def __init__(self, parent=None, in_color=QColor(255, 0, 0), mode="UK"):
        super(PyAnalogClock, self).__init__(parent)
        """
        :type parent: Nonetype
        :param parent:
        :type in_color: 
        :param in_color:
        :type mode: string
        :param mode:
        :rtype: None
        """

        # Change timezones "AUS" "MOS" "UK" "LA"
        self.clock_face = "" 
        if (mode=="AUS"):
            self.clock_face = "AUS"
        elif (mode=="MOS"):
            self.clock_face = "MOS"
        elif (mode=="UK"):
            self.clock_face = "UK"
        elif (mode=="LA"):
            self.clock_face = "LA"

        self.timeZoneOffset = 0

        # clock graphic face
        timer = QtCore.QTimer(self)
        timer.timeout.connect(self.update)
        timer.timeout.connect(self.updateTime)
        timer.start(1000)

        self.setWindowTitle(QtCore.QObject.tr(self, "Analog Clock"))
        self.resize(200, 200)

        self.hourHand = QtGui.QPolygon([
            QtCore.QPoint(7, 8),
            QtCore.QPoint(-7, 8),
            QtCore.QPoint(0, -40)
        ])
        self.minuteHand = QtGui.QPolygon([
            QtCore.QPoint(7, 8),
            QtCore.QPoint(-7, 8),
            QtCore.QPoint(0, -70)
        ])

        self.hourColor = in_color
        self.minuteColor = in_color
        self.faceColor = QColor(255, 255, 255)

    def paintEvent(self, event):
        '''
        :type paintEvent: function
        :param paintEvent: displays the clock face graphic
        '''
        side = min(self.width(), self.height())
        time = QDateTime.currentDateTime()
        time = QDateTime.toUTC(time)

        time = self.timezone_population(time, mode=self.clock_face)
        # could we set the tool tip here for the month day and time in digital maybe?
        self.setToolTip(self.digital_clock())

        timeh = QDateTime.toString(time,"hh")
        timeh = int(timeh)
        
        timem = QDateTime.toString(time,"mm")
        timem = int(timem)

        times = QDateTime.toString(time,"ss")
        times = int(times)

        painter = QtGui.QPainter()
        painter.begin(self)
        painter.setRenderHint(QtGui.QPainter.Antialiasing)
        painter.translate(self.width() / 2, self.height() / 2)
        painter.scale(side / 200.0, side / 200.0)

        painter.setPen(QtCore.Qt.NoPen)
        painter.setBrush(QtGui.QBrush(self.hourColor))

        painter.save()
        painter.rotate(30.0 * ((timeh + timem / 60.0)))
        painter.drawConvexPolygon(self.hourHand)
        painter.restore()

        painter.setPen(self.faceColor)

        for i in range(0, 12):
            painter.drawLine(75, 0, 96, 0)
            painter.rotate(30.0)

        painter.setPen(QtCore.Qt.NoPen)
        painter.setBrush(QtGui.QBrush(self.minuteColor))

        painter.save()
        painter.rotate(6.0 * (timem + times / 60.0))
        painter.drawConvexPolygon(self.minuteHand)
        painter.restore()

        painter.setPen(QtGui.QPen(self.faceColor))

        for j in range(0, 60):
            if (j % 5) != 0:
                painter.drawLine(90, 0, 96, 0)
            painter.rotate(6.0)

        painter.end()

    def updateTime(self):
        """
        :type self: 
        :param self:
        :rtype: None
        """
        # call your function pass in self.time_zone
        self.timeChanged.emit(QtCore.QTime.currentTime())

    def timezone_population(self, time, mode="UK"):
        '''
        :type timezone_population: function
        :param timezone_population: adds seconds onto the UTZ time to generate
        different timezones
        :rtype: integer
        '''
        # To add a timezone, add GMT hours * 60(minutes) * 60(seconds) 
        # to get GMT seconds 1 hour = 3600 seconds
        x = 0
        y = 3600 #dst + or -
        if (mode=="UK"):
            x = 3600
        elif (mode=="AUS"):
            x = 36000
        elif (mode=="MOS"):
            x = 10800
        elif (mode=="LA"):
            x = -25200
        time = time.addSecs(x)
        return time

    def digital_clock(self):
        '''
        :type digital_clock: function
        :param digital_clock: determines UTC time and also current time of
        local computer. Displays time in format yyyy/MM/dd hh:mm, this function
        helps the clock determine the digital time format.
        :rtype: integer
        '''
        side = min(self.width(), self.height())
        time = QDateTime.currentDateTime()
        time = QDateTime.toUTC(time)
        time = self.timezone_population(time, mode=self.clock_face)
        time = QDateTime.toString(time,"yyyy/MM/dd\nhh:mm ap")
        return time
