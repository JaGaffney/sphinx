[docs]    def __init__(self, current_show, in_context=None):
        super(showInfo, self).__init__()
        """
        :type current_show: string
        :param current_show: the current show name being selected from the main ui program
        :type in_context: class
        :param in_context: uses the context.py file if context was not called correctly
        :rtype: None
        """

            self.context = in_context
        else:    
            self.context = CsContext()

        #  pass in the context to the scdeule manager also

        ####################################################################
        # grab upfront user data by populating show info tab with json file
        ####################################################################
        self.show_json = self.schedule_manager.show_json

        # correct missing entries by creating blank defaults
        if ("show_info" not in list(self.show_json[current_show].keys())):
            self.show_json[current_show]["show_info"] = cinestron_task_info.show_info_default
        if ("resolution" not in list(self.show_json[current_show]["show_info"].keys())):
            self.show_json[current_show]["show_info"]["resolution"] = ""
        if ("frame_length" not in list(self.show_json[current_show]["show_info"].keys())):
            self.show_json[current_show]["show_info"]["frame_length"] = ""
        if ("info" not in list(self.show_json[current_show]["show_info"].keys())):
            self.show_json[current_show]["show_info"]["info"] = ""
        if ("client" not in list(self.show_json[current_show]["show_info"].keys())):
            self.show_json[current_show]["show_info"]["client"] = ""
        if ("contact" not in list(self.show_json[current_show]["show_info"].keys())):
            self.show_json[current_show]["show_info"]["contact"] = ""

        self.active_resolution = self.show_json[current_show]["show_info"]["resolution"] 
        self.active_shot_time = self.show_json[current_show]["show_info"]["frame_length"]

        self.active_client_input = self.show_json[current_show]["show_info"]["client"]
        self.active_contact_input = self.show_json[current_show]["show_info"]["contact"]

[docs]    def add_show_info(self, current_show, resolution, shot_time, client, contact, text):
        """
        :type current_show: string
        :param current_show: gathers the data from the QLineedit to be set into the dict
        :type resolution: string
        :param resolution: gathers the data from the QLineedit to be set into the dict
        :type shot_time: integer
        :param shot_time: gathers the data from the QLineedit to be set into the dict
        :type client: string
        :param client: gathers the data from the QLineedit to be set into the dict
        :type contact: string
        :param contact: gathers the data from the QLineedit to be set into the dict
        :type text: string
        :param text: gathers the data from the QLineedit to be set into the dict
        :rtype: None
        """
        self.active_resolution = resolution
        self.active_shot_time = shot_time
        self.active_additional_text = text
        self.active_client_input = client
        self.active_contact_input = contact
        self.show_json[current_show]["show_info"]["resolution"] = self.active_resolution
        self.show_json[current_show]["show_info"]["frame_length"] = self.active_shot_time
        self.show_json[current_show]["show_info"]["info"] = self.active_additional_text
        self.show_json[current_show]["show_info"]["client"] = self.active_client_input
        self.show_json[current_show]["show_info"]["contact"] = self.active_contact_input
        self.schedule_manager.localbd_push(self.show_json)
        self.close()
