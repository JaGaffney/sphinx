def modification_date(filename):
    t = os.path.getmtime(filename)
    my_date = datetime.datetime.fromtimestamp(t)
    return_date = ('%02d' % my_date.month)
    return_date += ('-%02d' % my_date.day)
    return_date += ('-%s' % str(my_date.year)[2] + str(my_date.year)[3])
    return_date += ('-%s' % my_date.hour)
    return_date += ('_%s' % my_date.minute)
    return return_date

    def add_to_recursion(self, file_directories_total, my_callback):
        for keys in file_directories_total.keys():
            
            # early out
            if (self.G_THREAD_RUNNING == False):
                return

            current_item = file_directories_total[keys]
            if (str(current_item["file_type"]) != str("dir") and current_item["hidden"] == False):
                message = ""
                # check our behvaiours
                mode = None 
                if (current_item["status"] == "red"):
                    mode = "pull"
                elif (current_item["status"] == "blue"):
                    mode = "push"
                elif (current_item["status"] == "orange"):
                    if (current_item["status_winner"] == "server"):
                        mode = "pull"
                    elif (current_item["status_winner"] == "local"):
                        mode = "push"
                elif (current_item["status"] == "green"):
                    mode = "match"

                #
                if (mode == "pull"):
                    message = "\tPull:" + current_item["file"]
                    message += " of size " + str(current_item["server_size"])
                    self.recursive_size += int(current_item["server_bytes"])
                    self.recursive_size_pull += int(current_item["server_bytes"])
                    my_item = QListWidgetItem(message)
                    my_item.setForeground(QColor(250, 0, 0))
                    my_callback.emit(my_item)
                elif (mode == "push"):
                    message = "\tPush:" + current_item["file"]
                    message += " of size " + str(current_item["local_size"])
                    self.recursive_size += int(current_item["local_bytes"])
                    self.recursive_size_push += int(current_item["local_bytes"])
                    my_item = QListWidgetItem(message)
                    my_item.setForeground(QColor(42,130,218))
                    my_callback.emit(my_item)
                elif (mode == "match"):
                    message = "\tmatch:" + current_item["file"]
                    message += " of size " + str(current_item["local_size"])
                    self.recursive_size += int(current_item["local_bytes"])
                    self.recursive_size_match += int(current_item["local_bytes"])
                    my_item = QListWidgetItem(message)
                    my_item.setForeground(QColor(0,250,0))
                    my_callback.emit(my_item)
                #
                self.recursive_results.append(current_item)
            elif (str(current_item["file_type"]) == str("dir") and current_item["hidden"] == False):
                self.recursive_results_dirs.append(current_item)
