[docs]    def rename_dict(self, old_name, new_name):
        """
        :type old_name: string
        :param old_name: old key name
        :type new_name: string
        :param new_name: new key name that was input into the QLineEdit()
        :rtype: None
        """
        mode = ""
        in_dict = self.schedule_manager.show_json
        new_name = str(new_name)
        show = self.schedule_manager.active_show
        sequence = self.schedule_manager.active_sequence
        shot = self.schedule_manager.active_shot

        if (show != "" and sequence == "" and shot ==""):
            in_dict[old_name]["show_name"] = new_name
            in_dict[new_name] = in_dict[old_name]
            in_dict.pop(old_name)
            mode = "show"
        elif (show != "" and sequence != "" and shot == ""):
            in_dict[show][new_name] = in_dict[show][old_name]
            in_dict[show].pop(old_name)
            mode = "sequence"
        elif (show != "" and sequence != "" and shot != ""):
            in_dict[show][sequence][new_name] = in_dict[show][sequence][old_name]
            in_dict[show][sequence].pop(old_name)
            mode = "shot"

        self.schedule_manager.localbd_push(in_dict)

        if (mode == "show"):
            self.schedule_manager.refresh_show()
        elif (mode == "sequence"):
            self.schedule_manager.refresh_sequence()
        elif (mode == "shot"):
            self.schedule_manager.refresh_shot() 

        self.close()

    def duplicate_dict(self, old_name, new_name):
        """
        :type old_name: string
        :param old_name: old key name
        :type new_name: string
        :param new_name: new key name that was input into the QLineEdit()
        :rtype: None
        """
        show = self.schedule_manager.active_show
        sequence = self.schedule_manager.active_sequence
        shot = self.schedule_manager.active_shot
        in_dict = self.schedule_manager.show_json
        new_name = str(new_name)
        mode = ""

        if (show !="" and sequence != "" and shot == ""):
            temp_duplicate = in_dict[show][old_name]
            in_dict[show][new_name] = temp_duplicate
            mode = "sequence"
        elif (show !="" and sequence != "" and shot != ""):
            temp_duplicate = in_dict[show][sequence][old_name]
            in_dict[show][sequence][new_name] = temp_duplicate
            mode = "shot"

        # build the json structure
        self.schedule_manager.localbd_push(in_dict) 

        # refreshes UI list
        if (mode == "sequence"):
            self.schedule_manager.refresh_sequence()
        elif (mode == "shot"):
            self.schedule_manager.refresh_shot()           
    
        self.close()

    def create_calender(self):
        """
        :type self: 
        :param self: creates the ui for the calender widget
        :rtype: object
        """
        window_name = "Calendar"
        self.setWindowTitle(window_name)
        self.cal = QCalendarWidget(self)
        self.cal.clicked.connect(self.update_date)
        self.resize(300, 300)
        self.cal.resize(300, 300)

    def get_date(self):
        """
        :type self: 
        :param self: sets the date as a string
        :rtype: string
        """
        self.active_date = self.update_date()
        return self.active_date
 
    def update_date(self,*args):
        """
        :type self: 
        :param self: gets the date that is selecetd in the calander widget
        :rtype: string
        """
        self.active_date = self.cal.selectedDate().toString()
        #self.cal.deleteLater()
        self.close()
        return self.active_date
