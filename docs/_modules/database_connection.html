def assure_path_exists_server(currentDir, ftp, first_run=True):
    """
    :type currentDir: string
    :param currentDir: finds the current directory of the server
    :type ftp: string
    :param ftp: finds the current working directory of the server
    :type first_run: mode
    :param first_run: mode to check if its first run
    :rtype: None
    """
    #
    if (first_run == True):
        currentDir = currentDir.replace(ftp.local_root, ftp.server_root)
    #
    if currentDir != "":
        try:
            ftp.ftp.cwd(currentDir)
        except:
            try:
                ftp.ftp.mkd(currentDir)
                print("\nmade server directory: (%s)" % currentDir)
            except:
                do_nothing = 0

    def db_get(self, first_run=False):
        """
        :type first_run: boolean
        :param first_run: gathers the data from the mongodb server when active
        :rtype: dict

        # server initlize this first
        # get from the mongoDB
        if (do_mongodb==True):
            self.show_json = self.mongodb_get(do_mongodb=True)
            self.show_json.pop("_id")
        # if mongoDB failes get from the server
        if (self.show_json == {}):
            self.ftpdb_get(first_run=first_run)
        # if server + mongoDB have failed, get from local
        if (self.show_json == {}):
            self.show_json = read_json(self.context.get_master_json())
        
        # 
        return self.show_json

    def db_push(self, in_json, sync_ftp=False):
        """
        :type in_json: dict
        :param in_json: pushes the data from the dict to the ftpserver
        :rtype: dict
        """
        # save to mongo db
        if (do_mongodb==True):
            self.mongodb_push(in_json)
        else:
            # if mongo fails allways sync to the ftp
            sync_ftp = True

        if (sync_ftp==True):
            # then save to the ftp server
            self.ftpdb_push(in_json)

            # read the server json
            server_json_path = os.path.join(self.orig_location, self.directory)
            self.server_json = read_json(server_json_path)

            # straight download at the moment no merging
            # push back
            #if (self.show_json == {}):
            #    self.show_json = self.server_json
            self.show_json = self.server_json

            # this doenst really need to be called everytime
            if (first_run==True):
                # overwrite local
                self.save_dict(self.show_json)

    def mongodb_get(self, do_mongodb=False):
        """
        :type do_mongodb: boolean
        :param do_mongodb: checks the context file to see if database connection is turned on
        :rtype: dict
        """
        if (do_mongodb == True):
            print("Connecting to mongodb server...")
            try:
                #gathers the data from the server
                project_dict = self.shows.find()

                #turns the data from the server into a python dict
                python_dict = {}
                for item in project_dict:
                    python_dict = item

                #find the unique id so that the self.mongodb_push() can save the data to the correct location
                self.primary_key = python_dict["_id"]
                return python_dict

            except:
                print("FAILED to connect to mongodb server")
                do_mongodb = False
                python_dict ={}
                return python_dict
        else:
            do_mongodb = False
            python_dict ={}
            return python_dict

    def mongodb_push(self, in_dict):
        """
        :type in_dict: dict

        :rtype: None

        if in_dict != {}:
            #
            self.save_dict(in_dict) 

            if (do_mongodb == True):
                try:
                    project_dict = self.shows.find()

                    #turns the data from the server into a python dict
                    python_dict = {}
                    for item in project_dict:
                        python_dict = item
                    #find the unique id so that the self.mongodb_push() can save the data to the correct location
                    self.primary_key = python_dict["_id"]

                    # updates the dict by saving at the location of the current primary key

            print("\nno data inside dictionary\n")

    def create_folders( self,
                        d,
                        current_dir='../../SHOWS',
                        build_folders=False,
                        build_server_folders=False):
        """
        :type d: dict
        :param d: the show_json dictionary which keys will be used for dir creation
        :type current_dir: string
        :param current_dir: finds the location of where the folders will be created
        :type build_folders: boolean
        :param build_folders: folder building will only happen when applicable due to lag when creating
        :type build_server_folders: boolean
        :param build_server_folders: folder building will only happen when applicable due to lag when creating
        :rtype: None
        """
        if type(d) is dict:
            for key, val in list(d.items()):
                if key in non_create_keys:
                    continue

                # mechanisms to stop building hidden folder and subfolders.
                # by still using the recursion design pattern
                hidden = False
                if type(val) is dict:
                    # check for hidden one down to stop building if found
                    if ("hidden" in list(val.keys())):
                        tmp_data = val["hidden"] 
                        if (tmp_data == True or tmp_data == "True" or tmp_data == "true"):
                            hidden = True
                    # check for active to stop building task folders
                    if ("active" in list(val.keys())):
                        tmp_data = val["active"] 
                        if (tmp_data == "" or tmp_data == False or tmp_data == "False" or tmp_data == "false"):
                            hidden = True

                # only build valid folders
                #
                if (hidden == False):
                    if (self.context.get_debug()):
                        print("key")
                        print(key)

                    if (key != "task_data"):
                        #
                        make_dir = os.path.join(current_dir, key)
                        if not os.path.exists(make_dir):
                            if (self.context.get_debug()):
                                print("make_dir: %s" % make_dir)

                            os.mkdir(make_dir)

                        # 
                        if (build_server_folders==True):
                            assure_path_exists_server(make_dir, self.ftp)

                        if type(val) == dict or type(val) == list:
                            self.create_folders(val,
                                                os.path.join(current_dir, key),
                                                build_folders=build_folders,
                                                build_server_folders=build_server_folders)
                        #
                    else:
                        if type(val) == dict or type(val) == list:
                            self.create_folders(val,
                                                current_dir,  
                                                build_folders=build_folders,
                                                build_server_folders=build_server_folders)    
                #
            #
        else:
            #
